# Define query_player compatible with our notebook's API
def query_player_notebook(game, state):
    """Make a move by querying standard input."""
    print("\ncurrent state:")
    game.display(state)
    print("available moves: {}".format(game.actions(state)))
    print("")
    move = None
    if game.actions(state):
        move_string = input('Your move? ')
        try:
            move = eval(move_string)
        except NameError:
            move = move_string
    else:
        print('no legal moves: passing turn to next player')
    return move

MCTS vs Heuristic Alpha-Beta Search

Now let's test Monte Carlo Tree Search against the heuristic alpha-beta search player over 5 rounds.

# Import MCTS player from games4e
from games4e import monte_carlo_tree_search

# Create MCTS player wrapper
def mcts_player(game, state):
    """MCTS player that returns just the move (not the full action)."""
    return monte_carlo_tree_search(state, game, N=1000)

# Run 5 rounds: MCTS (X) vs Heuristic Alpha-Beta (O)
print("Running 5 rounds: MCTS (X) vs Heuristic Alpha-Beta (O)")
print("="*60)

results = {'X': 0, 'O': 0, 'Draw': 0}

for round_num in range(1, 6):
    print(f"\n{'='*60}")
    print(f"ROUND {round_num}")
    print(f"{'='*60}")
    
    result = play_game(ConnectFour(), 
                       dict(X=mcts_player, O=player(h_alphabeta_search)), 
                       verbose=False)
    
    if result.utility == 1:
        winner = 'X (MCTS)'
        results['X'] += 1
    elif result.utility == -1:
        winner = 'O (Alpha-Beta)'
        results['O'] += 1
    else:
        winner = 'Draw'
        results['Draw'] += 1
    
    print(f"\nRound {round_num} Result: {winner}")
